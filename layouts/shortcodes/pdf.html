{{- /* Shortcode to generate a base64-encoded data URI for a PDF for use in markdown content. This embeds the PDF directly in the HTML, avoiding caching. Usage in markdown: [Download PDF]({{< pdf "pdf/CGV.pdf" >}}) Parameters: .Get 0: Path to PDF file relative to assets/ directory */ -}} {{- $path := .Get 0 -}} {{- $pdf := resources.Get $path -}} {{- if $pdf -}} {{- printf "data:application/pdf;base64,%s" ($pdf.Content | base64Encode) -}} {{- else -}} {{- errorf "PDF file not found: %s" $path -}} {{- end -}}
