{{- /* Returns a base64-encoded data URI for a PDF file from the assets directory. This embeds the PDF directly in the HTML, avoiding caching. Usage: {{ partial "pdf-url.html" (dict "path" "pdf/CV-SAINTHILLIER-Guillaume.pdf") }} Parameters: - path: Path to PDF file relative to assets/ directory (required) */ -}} {{- $path := .path -}} {{- $pdf := resources.Get $path -}} {{- if $pdf -}} {{- printf "data:application/pdf;base64,%s" ($pdf.Content | base64Encode) -}} {{- else -}} {{- errorf "PDF file not found: %s" $path -}} {{- end -}}
