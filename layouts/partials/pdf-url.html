{{- /* Returns a fingerprinted permalink for a PDF file from the assets directory. Usage: {{ partial "pdf-url.html" (dict "path" "pdf/CV-SAINTHILLIER-Guillaume.pdf") }} Parameters: - path: Path to PDF file relative to assets/ directory (required) */ -}} {{- $path := .path -}} {{- $pdf := resources.Get $path -}} {{- if $pdf -}} {{- $pdfProcessed := $pdf | resources.Fingerprint "sha256" -}} {{- $pdfProcessed.Permalink -}} {{- else -}} {{- errorf "PDF file not found: %s" $path -}} {{- end -}}
